10) Min and Max

 function arrayMin(arr) {
            let min=arr[0];
           for(let i=0;i< arr.length;i++){
             if(arr[i]< min){
               min = arr[i];
             }
             
           }
           return min;
           
         }
         
         function arrayMax(arr) {
          let max=arr[0];
           for(let i=0;i < arr.length;i++){
             if(arr[i]>max){
               max = arr[i];
             }
             
           }
           return max;
         
         }

function Birthday_Game(arr,D ,M) {
            let N =arr.length
                   let count =0
                   
                   for(i=0; i< N-M+1; i++){
